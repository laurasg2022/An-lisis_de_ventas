def analizar_top_bottom_tienda(df, nombre_tienda):
    # Contar productos vendidos
    ventas = df['Producto'].value_counts().reset_index()
    ventas.columns = ['Producto', 'Cantidad Vendida']

    top5 = ventas.head(5)
    bottom5 = ventas.tail(5)

    print(f"\nüìç {nombre_tienda}")
    print("üîù Top 5 m√°s vendidos:")
    print(top5.to_string(index=False))

    print("\nüîª Bottom 5 menos vendidos:")
    print(bottom5.to_string(index=False))

    # Visualizaci√≥n conjunta en subplots
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))

    sns.barplot(data=top5, x='Cantidad Vendida', y='Producto', ax=axes[0], palette='crest')
    axes[0].set_title('Top 5 m√°s vendidos')

    sns.barplot(data=bottom5, x='Cantidad Vendida', y='Producto', ax=axes[1], palette='rocket')
    axes[1].set_title('Bottom 5 menos vendidos')

    for ax in axes:
        ax.set_xlabel('Cantidad Vendida')
        ax.set_ylabel('Producto')

    fig.suptitle(f'An√°lisis de ventas en {nombre_tienda}', fontsize=14)
    plt.tight_layout(rect=[0, 0, 1, 0.95])
    plt.show()

üìå Introducci√≥n¬∂

El objetivo principal es identificar cu√°l de las tiendas presenta un rendimiento comparativamente m√°s bajo y, por tanto, podr√≠a ser considerada para su venta, liberando as√≠ recursos que permitan al propietario invertir en un nuevo emprendimiento.

El an√°lisis se realiza teniendo en cuenta los siguientes indicadores clave:
